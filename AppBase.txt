          <svg v-if="saving" class="w-4 h-4 animate-spin" fill="none" viewBox="0 0 24 24">
            <circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4" />
            <path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8v8z" />
          </svg>
          {{ saving ? 'Guardando...' : 'Guardar CV' }}
        </button>
      </div>
    </div>

    <!-- Toast -->
    <Transition name="toast">
      <div v-if="toast.show" :class="['fixed bottom-6 right-6 z-50 flex items-center gap-2.5 px-4 py-3 rounded-xl text-sm font-medium shadow-xl border',
        toast.type === 'success'
          ? 'bg-slate-800 border-slate-700 text-white'
          : 'bg-red-950 border-red-800 text-red-200']">
        {{ toast.type === 'success' ? '✓' : '⚠' }} {{ toast.message }}
      </div>
    </Transition>

  </div>
</template>

<script setup>
import { ref, reactive, onMounted } from 'vue'
import { loadCV, saveCV, clearCV } from './db.js'
import BaseInput from "@/components/BaseInput.vue";
import {
  personalSchema,
  educacionItemSchema,
  experienciaItemSchema,
  proyectoItemSchema,
  certificacionItemSchema,
  validateSection,
  validateArray,
} from '@/Cvschema.js'

// ── Constants ──────────────────────────────────────────────
const levelLabels = ['', 'Básico', 'Elemental', 'Intermedio', 'Avanzado', 'Experto']

const inputClass = {
  class: [
    'w-full bg-slate-800/60 border border-slate-700 rounded-lg px-3 py-2.5',
    'text-sm text-slate-100 placeholder:text-slate-500',
    'focus:outline-none focus:ring-2 focus:ring-blue-500/50 focus:border-blue-500',
    'disabled:opacity-40 disabled:cursor-not-allowed transition-all',
  ].join(' ')
}

const defaultItems = {
  educacion: () => ({ institucion: '', titulo: '', area: '', estado: '', inicio: '', fin: '', descripcion: '' }),
  experiencia: () => ({ empresa: '', cargo: '', modalidad: '', tipo: '', inicio: '', fin: '', actual: false, descripcion: '' }),
  proyectos: () => ({ nombre: '', rol: '', url: '', repo: '', tecnologias: '', descripcion: '' }),
  certificaciones: () => ({ nombre: '', organizacion: '', fecha: '', expiracion: '', noExpira: false, url: '' }),
}

const makeDefaultData = () => ({
  personal: { nombre: '', apellido: '', email: '', telefono: '', ciudad: '', pais: '', linkedin: '', website: '', titulo: '', resumen: '' },
  educacion: [],
  experiencia: [],
  habilidades: { tecnicas: [], blandas: [], idiomas: [] },
  proyectos: [],
  certificaciones: [],
})

// ── State ──────────────────────────────────────────────────
const data = reactive(makeDefaultData())
const saving = ref(false)
const lastSaved = ref(null)
const newSoftSkill = ref('')
const toast = reactive({ show: false, message: '', type: 'success' })
let autoSaveTimer = null

// Validation errors
const personalErrors = ref({})
const educacionErrors = ref([])
const experienciaErrors = ref([])
const proyectosErrors = ref([])
const certificacionesErrors = ref([])

// ── IndexedDB via Dexie ────────────────────────────────────
onMounted(async () => {
  try {
    const stored = await loadCV()
    if (stored) Object.assign(data, { ...makeDefaultData(), ...stored })
  } catch (e) {
    showToast('Error al cargar los datos guardados', 'error')
  }
})

async function handleSave() {
  // Validate all sections with Zod
  const pv = validateSection(personalSchema, data.personal)
  personalErrors.value = pv.errors
  educacionErrors.value = validateArray(educacionItemSchema, data.educacion)
  experienciaErrors.value = validateArray(experienciaItemSchema, data.experiencia)
  proyectosErrors.value = validateArray(proyectoItemSchema, data.proyectos)
  certificacionesErrors.value = validateArray(certificacionItemSchema, data.certificaciones)

  const hasErrors =
    !pv.success ||
    educacionErrors.value.some(e => Object.keys(e).length > 0) ||
    experienciaErrors.value.some(e => Object.keys(e).length > 0) ||
    proyectosErrors.value.some(e => Object.keys(e).length > 0) ||
    certificacionesErrors.value.some(e => Object.keys(e).length > 0)

  if (hasErrors) {
    showToast('Corregí los errores antes de guardar', 'error')
    return
  }

  saving.value = true
  try {
    await saveCV(JSON.parse(JSON.stringify(data)))
    lastSaved.value = new Date().toLocaleTimeString('es-AR', { hour: '2-digit', minute: '2-digit' })
    showToast('CV guardado correctamente ✓', 'success')
  } catch (e) {
    showToast('Error al guardar en IndexedDB', 'error')
  } finally {
    saving.value = false
  }
}

// Silent autosave (no validation, no toast)
function autoSave() {
  clearTimeout(autoSaveTimer)
  autoSaveTimer = setTimeout(async () => {
    try {
      await saveCV(JSON.parse(JSON.stringify(data)))
      lastSaved.value = new Date().toLocaleTimeString('es-AR', { hour: '2-digit', minute: '2-digit' })
    } catch { }
  }, 1000)
}

async function handleClear() {
  if (!confirm('¿Borrar todos los datos del CV? Esta acción no se puede deshacer.')) return
  Object.assign(data, makeDefaultData())
  personalErrors.value = {}
  educacionErrors.value = []
  experienciaErrors.value = []
  proyectosErrors.value = []
  certificacionesErrors.value = []
  await clearCV()
  lastSaved.value = null
}

function showToast(message, type = 'success') {
  toast.message = message
  toast.type = type
  toast.show = true
  setTimeout(() => { toast.show = false }, 3500)
}

// ── List helpers ───────────────────────────────────────────
function addItem(section) {
  if (section in defaultItems) {
    data[section].push(defaultItems[section]())
    autoSave()
  }
}

function removeItem(section, index) {
  if (section.includes('.')) {
    const [parent, child] = section.split('.')
    data[parent][child].splice(index, 1)
  } else {
    data[section].splice(index, 1)
  }
  autoSave()
}

// ── Skills ─────────────────────────────────────────────────
function addSkill(type) {
  data.habilidades[type].push({ nombre: '', nivel: levelLabels[1], nivel_num: 1 })
  autoSave()
}

function removeSkill(type, index) {
  data.habilidades[type].splice(index, 1)
  autoSave()
}

function addSoftSkill() {
  const val = newSoftSkill.value.trim()
  if (val && !data.habilidades.blandas.includes(val)) {
    data.habilidades.blandas.push(val)
    newSoftSkill.value = ''
    autoSave()
  }
}

function removeSoftSkill(index) {
  data.habilidades.blandas.splice(index, 1)
  autoSave()
}
</script>
